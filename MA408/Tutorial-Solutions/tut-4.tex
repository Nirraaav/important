\section*{Setup}
Let $\mathcal{S} := \{(a, b] \mid -\infty \le a < b \le +\infty\} \cup \{\emptyset\}$. Let $F: \mathbb{R} \to \mathbb{R}$ be non-decreasing and right-continuous, extended to $\overline{\mathbb{R}}$ by limits. Define $\lambda_F((a, b]) = F(b) - F(a)$.

\hrulefill

\begin{problem}
Let $\{(a_i, b_i] \mid i=1, \cdots, k\} \subseteq \mathcal{S}$ be disjoint and let $\bigcup_{i=1}^k (a_i, b_i] \subseteq (a, b]$, where $a_i, b_i, a, b \in \overline{\mathbb{R}}$. Show that
\[ \sum_{i=1}^k \lambda_F((a_i, b_i]) \le \lambda_F((a, b]). \]
\end{problem}

\begin{solution}
Since the intervals $I_i = (a_i, b_i]$ are disjoint, we can reorder indices such that $a_1 < b_1 \le a_2 < b_2 \le \dots \le b_k$.
Since their union is contained in $(a, b]$, we have $a \le a_1$ and $b_k \le b$.
We consider the expansion of $F(b) - F(a)$ by inserting the points $a_i, b_i$:
\[ F(b) - F(a) = [F(a_1) - F(a)] + \sum_{i=1}^k [F(b_i) - F(a_i)] + \sum_{j=1}^{k-1} [F(a_{j+1}) - F(b_j)] + [F(b) - F(b_k)] \]
Since $F$ is non-decreasing:
\begin{itemize}
    \item $F(a_1) - F(a) \ge 0$ (since $a_1 \ge a$)
    \item $F(a_{j+1}) - F(b_j) \ge 0$ (since $a_{j+1} \ge b_j$)
    \item $F(b) - F(b_k) \ge 0$ (since $b \ge b_k$)
\end{itemize}
Let $S = \sum_{i=1}^k \lambda_F((a_i, b_i]) = \sum_{i=1}^k (F(b_i) - F(a_i))$.
The equation becomes:
\[ F(b) - F(a) = S + (\text{sum of non-negative terms}) \]
Therefore, $S \le F(b) - F(a)$, which means $\sum \lambda_F((a_i, b_i]) \le \lambda_F((a, b])$.
\end{solution}

\hrulefill

\begin{problem}
Suppose $[a, b] \subseteq \bigcup_{i=1}^k (a_i, b_i)$ with $a_i < b_i$ and $a, b, a_i, b_i \in \mathbb{R}$. Show that
\[ \sum_{i=1}^k (F(b_i) - F(a_i)) \ge F(b) - F(a). \]
\end{problem}

\begin{solution}
We construct a minimal chain of intervals covering $[a, b]$.
\begin{enumerate}
    \item Pick $i_1$ such that $a \in (a_{i_1}, b_{i_1})$. Thus $a_{i_1} < a < b_{i_1}$.
    \item If $b_{i_1} \le b$, pick $i_2$ such that $b_{i_1} \in (a_{i_2}, b_{i_2})$. Thus $a_{i_2} < b_{i_1} < b_{i_2}$.
    \item Repeat until we pick $i_m$ such that $b_{i_m} > b$.
\end{enumerate}
Consider the sum over this subsequence:
\[ S' = \sum_{j=1}^m (F(b_{i_j}) - F(a_{i_j})) \]
Rearranging terms:
\[ S' = F(b_{i_m}) - F(a_{i_1}) + \sum_{j=1}^{m-1} (F(b_{i_j}) - F(a_{i_{j+1}})) \]
Using the property that $b_{i_j} > a_{i_{j+1}}$ (from the overlap in the chain construction) and $F$ is non-decreasing, we have $F(b_{i_j}) - F(a_{i_{j+1}}) \ge 0$.
Also $b_{i_m} > b \implies F(b_{i_m}) \ge F(b)$ and $a_{i_1} < a \implies F(a_{i_1}) \le F(a) \implies -F(a_{i_1}) \ge -F(a)$.
Thus:
\[ S' \ge F(b) - F(a) \]
Since the total sum includes $S'$ and all terms are non-negative:
\[ \sum_{i=1}^k (F(b_i) - F(a_i)) \ge F(b) - F(a). \]
\end{solution}

\begin{problem}
Consider the Borel $\sigma$-algebra $\mathcal{B}$ on $\mathbb{R}$. Suppose $\mu: \mathcal{B} \to [0, \infty]$ be an additive set function such that $\mu(I) < \infty$ for all bounded intervals $I \subset \mathbb{R}$. Show existence of non-decreasing $F$ such that $\mu((a, b]) = \lambda_F((a, b])$. Is $F$ unique? If $\mu$ is a measure, show $F$ is right continuous.
\end{problem}

\begin{solution}
\textbf{Existence:} Define $F: \mathbb{R} \to \mathbb{R}$ by fixing $F(0)=0$ and:
\[ F(x) = \begin{cases} \mu((0, x]) & x > 0 \\ -\mu((x, 0]) & x < 0 \end{cases} \]
For any $a < b$:
\begin{itemize}
    \item If $0 \le a < b$: $\mu((a, b]) = \mu((0, b] \setminus (0, a]) = \mu((0, b]) - \mu((0, a]) = F(b) - F(a)$.
    \item If $a < b \le 0$: $\mu((a, b]) = \mu((a, 0] \setminus (b, 0]) = \mu((a, 0]) - \mu((b, 0]) = -F(a) - (-F(b)) = F(b) - F(a)$.
    \item If $a < 0 < b$: $\mu((a, b]) = \mu((a, 0] \cup (0, b]) = \mu((a, 0]) + \mu((0, b]) = -F(a) + F(b)$.
\end{itemize}
Since $\mu \ge 0$, $F$ is non-decreasing.

\textbf{Uniqueness:} No. If $G$ satisfies the condition, $G(b) - G(a) = F(b) - F(a) \implies G(x) = F(x) + C$. $F$ is unique up to an additive constant.

\textbf{Right Continuity:} Suppose $\mu$ is a measure. Let $x_n \downarrow x$. The sequence of sets $E_n = (x, x_n]$ decreases to $\emptyset$. Since $\mu$ is finite on bounded intervals, by continuity of measure from above:
\[ \lim_{n \to \infty} \mu((x, x_n]) = \mu(\emptyset) = 0 \]
Substituting $F$:
\[ \lim_{n \to \infty} (F(x_n) - F(x)) = 0 \implies \lim_{n \to \infty} F(x_n) = F(x). \]
Thus, $F$ is right continuous.
\end{solution}
